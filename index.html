Connect to Google Sheets

Every RSVP will be logged as a new row in your spreadsheet — no coding needed, just copy & paste.

1
Create a new Google Sheet
Go to sheets.google.com → click Blank. Name it High Chai RSVPs.
2
Open Apps Script
In your Sheet, click Extensions → Apps Script. Delete any existing code in the editor.
3
Paste this script
Copy everything below and paste it into the Apps Script editor:
function doPost(e) {
  try {
    var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
    var data = JSON.parse(e.postData.contents);

    // Write headers on first submission
    if (sheet.getLastRow() === 0) {
      sheet.appendRow(data.headers);
      var hRange = sheet.getRange(1, 1, 1, data.headers.length);
      hRange.setBackground('#506552');
      hRange.setFontColor('#FFFFFF');
      hRange.setFontWeight('bold');
    }

    sheet.appendRow(data.values);
    sheet.autoResizeColumns(1, data.values.length);

    return ContentService
      .createTextOutput(JSON.stringify({ result: 'success' }))
      .setMimeType(ContentService.MimeType.JSON);
  } catch(err) {
    return ContentService
      .createTextOutput(JSON.stringify({ result: 'error', message: err.toString() }))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

function doGet(e) {
  return ContentService
    .createTextOutput(JSON.stringify({ status: 'High Chai RSVP endpoint is live!' }))
    .setMimeType(ContentService.MimeType.JSON);
}
4
Deploy as Web App
Click Deploy → New deployment. Set type to Web app. Under "Who has access" choose Anyone. Click Deploy and copy the Web App URL.
5
Paste the URL above
Go back to this page, paste the Web App URL into the green bar at the top, and click Connect ✓. You're done!
